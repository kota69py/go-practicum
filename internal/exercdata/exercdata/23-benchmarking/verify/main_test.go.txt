package main

import (
	"testing"
)

func TestConcat(t *testing.T) {
	vals := []string{"a", "b", "c"}
	r := Concat(vals)
	if r != "abc" {
		t.Errorf("Concat = %q, want %q", r, "abc")
	}
}

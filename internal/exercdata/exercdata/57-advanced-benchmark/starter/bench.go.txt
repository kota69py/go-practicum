func ConcatAppend(parts []string) string {
	result := ""
	for _, p := range parts {
		result += p
	}
	return result
}

func ConcatBuilder(parts []string) string {
	var b strings.Builder
	for _, p := range parts {
		b.WriteString(p)
	}
	return b.String()
}

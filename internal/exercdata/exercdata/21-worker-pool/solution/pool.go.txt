package main

import (
	"context"
	"fmt"
	"sync"
)

type Job struct {
	ID   int
	Data string
}

type Result struct {
	JobID  int
	Output string
	Err    error
}

func Worker(ctx context.Context, id int, jobs <-chan Job, results chan<- Result) {
	for {
		select {
		case <-ctx.Done():
			return
		case job, ok := <-jobs:
			if !ok {
				return
			}
			output := fmt.Sprintf("processed %s by worker %d", job.Data, id)
			results <- Result{JobID: job.ID, Output: output}
		}
	}
}

func StartPool(ctx context.Context, n int, jobs <-chan Job, results chan<- Result) {
	var wg sync.WaitGroup
	for i := range n {
		wg.Add(1)
		go func(id int) {
			defer wg.Done()
			Worker(ctx, id, jobs, results)
		}(i)
	}
	go func() {
		wg.Wait()
		close(results)
	}()
}

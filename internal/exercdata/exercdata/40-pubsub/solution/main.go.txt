package main

import (
	"fmt"
	"sync"
)

type Message struct {
	Topic string
	Data  interface{}
}

type PubSub struct {
	mu     sync.Mutex
	subs   map[string][]chan Message
	closed bool
}

func main() {
	fmt.Println("40-pubsub")
}

func New() *PubSub {
	return &PubSub{
		subs: make(map[string][]chan Message),
	}
}

func (ps *PubSub) Publish(topic string, data interface{}) {
	ps.mu.Lock()
	defer ps.mu.Unlock()
	if ps.closed {
		return
	}
	msg := Message{Topic: topic, Data: data}
	for _, ch := range ps.subs[topic] {
		ch <- msg
	}
}

func (ps *PubSub) Subscribe(topic string) <-chan Message {
	ps.mu.Lock()
	defer ps.mu.Unlock()
	ch := make(chan Message, 10)
	ps.subs[topic] = append(ps.subs[topic], ch)
	return ch
}

func (ps *PubSub) Close() {
	ps.mu.Lock()
	defer ps.mu.Unlock()
	if ps.closed {
		return
	}
	ps.closed = true
	for _, chs := range ps.subs {
		for _, ch := range chs {
			close(ch)
		}
	}
}

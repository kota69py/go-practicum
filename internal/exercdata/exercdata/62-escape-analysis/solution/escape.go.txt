package escape

import "testing"

func StackAlloc() int {
	x := 42
	return x
}

func HeapAlloc() *int {
	x := 42
	return &x
}

func InterfaceAlloc() interface{} {
	x := 42
	return x
}

func ClosureAlloc() func() {
	x := 42
	return func() {
		_ = x
	}
}

func SliceAllocFixed() []byte {
	return make([]byte, 1024)
}

func SliceAllocLarge() []byte {
	return make([]byte, 100000)
}

var global *int

func GlobalAlloc() *int {
	x := 42
	global = &x
	return global
}

func BenchmarkAlloc(b *testing.B, fn func()) testing.BenchmarkResult {
	return testing.Benchmark(func(b *testing.B) {
		for i := 0; i < b.N; i++ {
			fn()
		}
	})
}

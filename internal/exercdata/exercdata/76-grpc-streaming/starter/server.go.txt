package main

import (
	"fmt"
	"io"
	"log"
	"sync"

	pb "go-practicum-grpc-streaming/proto/stream"
	"google.golang.org/grpc"
)

type ChatRoom struct {
	mu      sync.Mutex
	members map[string]chan *pb.ChatMessage
}

func NewChatRoom() *ChatRoom {
	return &ChatRoom{members: make(map[string]chan *pb.ChatMessage)}
}

func (r *ChatRoom) Join(user string) chan *pb.ChatMessage {
	r.mu.Lock()
	defer r.mu.Unlock()
	ch := make(chan *pb.ChatMessage, 10)
	r.members[user] = ch
	return ch
}

func (r *ChatRoom) Leave(user string) {
	r.mu.Lock()
	defer r.mu.Unlock()
	if ch, ok := r.members[user]; ok {
		close(ch)
		delete(r.members, user)
	}
}

func (r *ChatRoom) Broadcast(sender string, msg *pb.ChatMessage) {
	r.mu.Lock()
	defer r.mu.Unlock()
	for user, ch := range r.members {
		if user != sender {
			select {
			case ch <- msg:
			default:
			}
		}
	}
}

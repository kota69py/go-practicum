package main

import (
	"context"
	"errors"
	"testing"

	"go.uber.org/mock/gomock"
)

func TestRegister_Success(t *testing.T) {
	ctrl := gomock.NewController(t)
	defer ctrl.Finish()

	mockRepo := NewMockUserRepository(ctrl)
	mockNotif := NewMockNotificationService(ctrl)

	svc := NewUserService(mockRepo, mockNotif)
	user := &User{ID: "1", Email: "test@example.com", Name: "Test"}

	mockRepo.EXPECT().Save(gomock.Any(), user).Return(nil)
	mockNotif.EXPECT().SendEmail(gomock.Any(), user.Email, "Welcome!", "Thanks for joining!").Return(nil)

	if err := svc.Register(context.Background(), user); err != nil {
		t.Fatalf("Register error: %v", err)
	}
}

func TestDeactivate_Success(t *testing.T) {

	defer ctrl.Finish()

	mockRepo := NewMockUserRepository(ctrl)
	mockNotif := NewMockNotificationService(ctrl)

	svc := NewUserService(mockRepo, mockNotif)

	mockRepo.EXPECT().Get(gomock.Any(), "1").Return(user, nil)

	if err := svc.Deactivate(context.Background(), "1"); err != nil {
		t.Fatalf("Deactivate error: %v", err)
	}
}

func TestRegister_Times(t *testing.T) {
	ctrl := gomock.NewController(t)
	defer ctrl.Finish()

	mockRepo := NewMockUserRepository(ctrl)
	mockNotif := NewMockNotificationService(ctrl)

	svc := NewUserService(mockRepo, mockNotif)

	mockRepo.EXPECT().Save(gomock.Any(), gomock.Any()).Return(nil).Times(2)
	mockNotif.EXPECT().SendEmail(gomock.Any(), gomock.Any(), gomock.Any(), gomock.Any()).Return(nil).Times(2)

	svc.Register(context.Background(), &User{ID: "1", Email: "a@b.com"})
	svc.Register(context.Background(), &User{ID: "2", Email: "c@d.com"})
}

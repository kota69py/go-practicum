package main

import (
	"net/http"
	"os"
	"testing"
	"time"
)

func TestRunServerStartsAndResponds(t *testing.T) {
	handler := http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.WriteHeader(http.StatusOK)
	})

	sig := make(chan os.Signal, 1)
	go func() {
		time.Sleep(50 * time.Millisecond)
		sig <- os.Interrupt
	}()

	err := RunServer(":0", handler, 5*time.Second, sig)
	if err != nil {
		t.Fatalf("RunServer error: %v", err)
	}
}

func TestRunServerImmediateSignal(t *testing.T) {
	handler := http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.WriteHeader(http.StatusOK)
	})

	sig := make(chan os.Signal, 1)
	close(sig) // signal already closed

	err := RunServer(":0", handler, 5*time.Second, sig)
	if err != nil {
		t.Fatalf("RunServer error: %v", err)
	}
}

func TestRunServerNilHandler(t *testing.T) {
	sig := make(chan os.Signal, 1)
	go func() {
		time.Sleep(50 * time.Millisecond)
		sig <- os.Interrupt
	}()

	err := RunServer(":0", nil, 5*time.Second, sig)
	if err != nil {
		t.Fatalf("RunServer with nil handler error: %v", err)
	}
}

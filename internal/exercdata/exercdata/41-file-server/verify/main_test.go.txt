package main

import (
	"net/http"
	"net/http/httptest"
	"os"
	"path/filepath"
	"testing"
)

func TestFileServer(t *testing.T) {
	t.Run("serves file from directory", func(t *testing.T) {
		tmpDir := t.TempDir()
		err := os.WriteFile(filepath.Join(tmpDir, "hello.txt"), []byte("world"), 0644)
		if err != nil {
			t.Fatal(err)
		}
		fs := NewFileServer(http.Dir(tmpDir))
		server := httptest.NewServer(fs)
		defer server.Close()

		resp, err := http.Get(server.URL + "/hello.txt")
		if err != nil {
			t.Fatal(err)
		}
		defer resp.Body.Close()
		if resp.StatusCode != http.StatusOK {
			t.Errorf("status = %d, want %d", resp.StatusCode, http.StatusOK)
		}
	})

	t.Run("returns 404 for missing file", func(t *testing.T) {
		fs := NewFileServer(http.Dir(t.TempDir()))
		server := httptest.NewServer(fs)
		defer server.Close()

		resp, _ := http.Get(server.URL + "/nope.txt")
		if resp.StatusCode != http.StatusNotFound {
			t.Errorf("status = %d, want %d", resp.StatusCode, http.StatusNotFound)
		}
	})
}

func TestStaticFileServer(t *testing.T) {
	tmpDir := t.TempDir()
	err := os.WriteFile(filepath.Join(tmpDir, "index.html"), []byte("<h1>hello</h1>"), 0644)
	if err != nil {
		t.Fatal(err)
	}
	handler := StaticFileServer(tmpDir)
	server := httptest.NewServer(handler)
	defer server.Close()

	resp, err := http.Get(server.URL + "/index.html")
	if err != nil {
		t.Fatal(err)
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		t.Errorf("status = %d, want %d", resp.StatusCode, http.StatusOK)
	}
}

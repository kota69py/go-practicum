package main

import (
	"context"
	"testing"
)

func TestNewPipeline(t *testing.T) {
	ctx := context.Background()
	gen := func(ctx context.Context) <-chan int {
		ch := make(chan int)
		go func() {
			defer close(ch)
			for i := 0; i < 5; i++ {
				ch <- i
			}
		}()
		return ch
	}
	p := NewPipeline(ctx, gen)
	if p == nil {
		t.Fatal("NewPipeline returned nil")
	}
}

func TestFanOut(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()
	gen := func(ctx context.Context) <-chan int {
		ch := make(chan int)
		go func() {
			defer close(ch)
			for i := 0; i < 3; i++ {
				ch <- i
			}
		}()
		return ch
	}
	p := NewPipeline(ctx, gen)
	worker := func(ctx context.Context, val int) int {
		return val * 2
	}
	outputs := p.FanOut(2, ctx, worker)
	if len(outputs) != 2 {
		t.Errorf("expected 2 output channels, got %d", len(outputs))
	}
	// Read from outputs to verify fan-out works
	var results []int
	for _, out := range outputs {
		for v := range out {
			results = append(results, v)
		}
	}
	if len(results) == 0 {
		t.Error("expected results from fan-out, got none")
	}
}

func TestFanIn(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	ch1 := make(chan int)
	ch2 := make(chan int)
	go func() {
		ch1 <- 1
		ch1 <- 2
		close(ch1)
	}()
	go func() {
		ch2 <- 3
		ch2 <- 4
		close(ch2)
	}()

	merged := FanIn(ctx, ch1, ch2)
	results := make(map[int]bool)
	for v := range merged {
		results[v] = true
	}
	for _, v := range []int{1, 2, 3, 4} {
		if !results[v] {
			t.Errorf("missing value %d in merged results", v)
		}
	}
}

func TestContextCancel(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())

	ch := make(chan int, 10)
	ch <- 1
	ch <- 2
	close(ch)

	cancel()

	merged := FanIn(ctx, ch)
	for v := range merged {
		_ = v // should just exit without hang
	}
}

func TestFanInEmpty(t *testing.T) {
	ctx := context.Background()
	ch := make(chan int)
	close(ch)

	merged := FanIn(ctx, ch)
	count := 0
	for range merged {
		count++
	}
	if count != 0 {
		t.Errorf("expected 0 values from empty channel, got %d", count)
	}
}

func TestFanInSingleChannel(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	ch := make(chan int, 5)
	for i := 0; i < 5; i++ {
		ch <- i * 10
	}
	close(ch)

	merged := FanIn(ctx, ch)
	var results []int
	for v := range merged {
		results = append(results, v)
	}
	if len(results) != 5 {
		t.Errorf("expected 5 values, got %d", len(results))
	}
}

import "C"
import (
	"fmt"
	"unsafe"
)

func Add(a, b int) int {
	return int(C.add(C.int(a), C.int(b)))
}

func Multiply(a, b float64) float64 {
	return float64(C.multiply(C.double(a), C.double(b)))
}

func StringLength(s string) int {
	cs := C.CString(s)
	defer C.free(unsafe.Pointer(cs))
	return int(C.string_length(cs))
}

func Concat(a, b string) string {
	ca := C.CString(a)
	cb := C.CString(b)
	defer C.free(unsafe.Pointer(ca))
	defer C.free(unsafe.Pointer(cb))
	result := C.concat(ca, cb)
	if result == nil {
		return ""
	}
	defer C.free(unsafe.Pointer(result))
	return C.GoString(result)
}

func main() {
	fmt.Println("CGO Demo")
	fmt.Printf("Add(3, 4) = %d\n", Add(3, 4))
	fmt.Printf("Multiply(2.5, 3.0) = %.1f\n", Multiply(2.5, 3.0))
	fmt.Printf("StringLength(\"hello\") = %d\n", StringLength("hello"))
	fmt.Printf("Concat(\"Hello \", \"World!\") = %s\n", Concat("Hello ", "World!"))
}

package main

import "testing"

func TestSortByLength(t *testing.T) {
	vals := []string{"aaa", "b", "cc"}
	SortByLength(vals)
	expected := []string{"b", "cc", "aaa"}
	for i, v := range vals {
		if v != expected[i] {
			t.Errorf("vals[%d] = %q, want %q", i, v, expected[i])
		}
	}
}

func TestSortByAge(t *testing.T) {
	people := []Person{
		{Name: "Alice", Age: 30},
		{Name: "Bob", Age: 25},
		{Name: "Charlie", Age: 30},
	}
	SortByAge(people)
	if people[0].Name != "Bob" {
		t.Errorf("first should be Bob, got %s", people[0].Name)
	}
}

func TestSearchInt(t *testing.T) {
	vals := []int{1, 3, 5, 7, 9}
	if !SearchInt(vals, 5) {
		t.Error("expected to find 5")
	}
	if SearchInt(vals, 4) {
		t.Error("expected NOT to find 4")
	}
	if SearchInt([]int{}, 1) {
		t.Error("expected NOT to find in empty")
	}
}

func TestDeduplicate(t *testing.T) {
	vals := []int{1, 1, 2, 3, 3, 3, 4, 5, 5}
	result := Deduplicate(vals)
	expected := []int{1, 2, 3, 4, 5}
	if len(result) != len(expected) {
		t.Fatalf("len = %d, want %d", len(result), len(expected))
	}
	for i, v := range result {
		if v != expected[i] {
			t.Errorf("result[%d] = %d, want %d", i, v, expected[i])
		}
	}
}

func TestDeduplicateEmpty(t *testing.T) {
	result := Deduplicate([]int{})
	if result == nil {
		t.Error("expected empty slice, not nil")
	}
	if len(result) != 0 {
		t.Errorf("len = %d, want 0", len(result))
	}
}

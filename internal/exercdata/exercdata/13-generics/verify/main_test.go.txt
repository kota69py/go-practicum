package main

import (
	"strings"
	"testing"
)

func TestMapIntToString(t *testing.T) {
	nums := []int{1, 2, 3}
	got := Map(nums, func(n int) string {
		return strings.Repeat("*", n)
	})
	want := []string{"*", "**", "***"}
	if len(got) != len(want) {
		t.Fatalf("got len %d, want %d", len(got), len(want))
	}
	for i := range got {
		if got[i] != want[i] {
			t.Errorf("got[%d] = %q, want %q", i, got[i], want[i])
		}
	}
}

func TestFilterEven(t *testing.T) {
	nums := []int{1, 2, 3, 4, 5, 6}

		t.Fatalf("got len %d, want %d", len(got), len(want))
	}
	for i := range got {
		if got[i] != want[i] {
			t.Errorf("got[%d] = %d, want %d", i, got[i], want[i])
		}
	}
}

func TestReduceSum(t *testing.T) {
	nums := []int{1, 2, 3, 4, 5}
	got := Reduce(nums, 0, func(acc, n int) int { return acc + n })
	if got != 15 {
		t.Errorf("got %d, want 15", got)
	}
}

func TestReduceConcat(t *testing.T) {
	words := []string{"hello", " ", "world"}
	got := Reduce(words, "", func(acc, s string) string { return acc + s })
	if got != "hello world" {
		t.Errorf("got %q, want %q", got, "hello world")
	}
}

func TestMapStringToUpper(t *testing.T) {
	words := []string{"go", "rust", "zig"}

		t.Fatalf("got len %d, want %d", len(got), len(want))
	}
	for i := range got {
		if got[i] != want[i] {
			t.Errorf("got[%d] = %q, want %q", i, got[i], want[i])
		}
	}
}

func TestFilterNoMatch(t *testing.T) {
	nums := []int{1, 3, 5}

}

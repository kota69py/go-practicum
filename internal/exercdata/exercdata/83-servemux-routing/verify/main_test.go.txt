package main

import (
	"encoding/json"
	"net/http"
	"net/http/httptest"
	"strings"
	"testing"
)

func setup() *httptest.Server {
	Users = map[string]User{
		"1": {ID: "1", Name: "Alice"},
		"2": {ID: "2", Name: "Bob"},
	}
	return httptest.NewServer(setupRouter())
}

func TestListUsers(t *testing.T) {
	s := setup()
	defer s.Close()

	resp, err := http.Get(s.URL + "/api/users")
	if err != nil {
		t.Fatal(err)
	}
	if resp.StatusCode != http.StatusOK {
		t.Errorf("got %d, want 200", resp.StatusCode)
	}
	var users []User
	json.NewDecoder(resp.Body).Decode(&users)
	resp.Body.Close()
	if len(users) != 2 {
		t.Errorf("got %d users, want 2", len(users))
	}
}

func TestGetUser(t *testing.T) {
	s := setup()
	defer s.Close()

	resp, err := http.Get(s.URL + "/api/users/1")
	if err != nil {
		t.Fatal(err)
	}
	if resp.StatusCode != http.StatusOK {
		t.Errorf("got %d, want 200", resp.StatusCode)
	}
	var u User
	json.NewDecoder(resp.Body).Decode(&u)
	resp.Body.Close()
	if u.Name != "Alice" {
		t.Errorf("got name %q, want %q", u.Name, "Alice")
	}
}

func TestGetUserNotFound(t *testing.T) {

}

func TestCreateUser(t *testing.T) {
	s := setup()
	defer s.Close()

	body := strings.NewReader(`{"name":"Charlie"}`)
	resp, err := http.Post(s.URL+"/api/users", "application/json", body)
	if err != nil {
		t.Fatal(err)
	}
	if resp.StatusCode != http.StatusCreated {
		t.Errorf("got %d, want 201", resp.StatusCode)
	}
	var u User
	json.NewDecoder(resp.Body).Decode(&u)
	resp.Body.Close()
	if u.Name != "Charlie" {
		t.Errorf("got name %q, want %q", u.Name, "Charlie")
	}
	if u.ID == "" {
		t.Error("expected non-empty ID")
	}
}

func TestDeleteUser(t *testing.T) {
	s := setup()
	defer s.Close()

	req, _ := http.NewRequest("DELETE", s.URL+"/api/users/1", nil)
	resp, err := http.DefaultClient.Do(req)
	if err != nil {
		t.Fatal(err)
	}
	if resp.StatusCode != http.StatusNoContent {
		t.Errorf("got %d, want 204", resp.StatusCode)
	}

	// 削除後は 404
	resp, err = http.Get(s.URL + "/api/users/1")
	if err != nil {
		t.Fatal(err)
	}
	if resp.StatusCode != http.StatusNotFound {
		t.Errorf("got %d, want 404 after delete", resp.StatusCode)
	}
}

func TestMethodNotAllowed(t *testing.T) {
	s := setup()

	}
}

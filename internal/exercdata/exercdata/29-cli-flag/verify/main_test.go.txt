package main

import (
	"os"
	"testing"
)

func TestGreetEnglish(t *testing.T) {
	got := greet("World", "en")
	if got != "Hello, World!" {
		t.Errorf("got %q, want %q", got, "Hello, World!")
	}
}

func TestGreetJapanese(t *testing.T) {
	got := greet("世界", "ja")
	if got != "こんにちは、世界！" {
		t.Errorf("got %q, want %q", got, "こんにちは、世界！")
	}
}

func TestGreetFrench(t *testing.T) {
	got := greet("Monde", "fr")
	if got != "Bonjour, Monde !" {
		t.Errorf("got %q, want %q", got, "Bonjour, Monde !")
	}
}

func TestAdd(t *testing.T) {
	if add(2, 3) != 5 {
		t.Errorf("2+3 = %d, want 5", add(2, 3))
	}
	if add(-1, 1) != 0 {
		t.Errorf("-1+1 = %d, want 0", add(-1, 1))
	}
}

func TestRunCLIGreet(t *testing.T) {
	os.Args = []string{"cli", "greet", "Alice"}
	RunCLI()
}

func TestRunCLIAdd(t *testing.T) {
	os.Args = []string{"cli", "add", "10", "20"}
	RunCLI()
}

func TestRunCLIHelp(t *testing.T) {
	os.Args = []string{"cli", "help"}
	RunCLI()
}

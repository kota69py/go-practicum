package main

import (
	"errors"
	"testing"
)

func TestParseUserOK(t *testing.T) {
	u, err := ParseUser("Alice,30")
	if err != nil {
		t.Fatalf("unexpected error: %v", err)
	}
	if u.Name != "Alice" {
		t.Errorf("Name = %q, want %q", u.Name, "Alice")
	}
	if u.Age != 30 {
		t.Errorf("Age = %d, want %d", u.Age, 30)
	}
}

func TestParseUserInvalidFormat(t *testing.T) {
	_, err := ParseUser("Alice")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
	if !errors.Is(err, ErrInvalidFormat) {
		t.Errorf("expected ErrInvalidFormat, got %v", err)
	}
}

func TestParseUserEmptyName(t *testing.T) {
	_, err := ParseUser(",30")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
	var pe *ParseError
	if !errors.As(err, &pe) {
		t.Errorf("expected *ParseError, got %v", err)
	}
	if pe.Message != "name is required" {
		t.Errorf("expected message %q, got %q", "name is required", pe.Message)
	}
}

func TestParseUserInvalidAge(t *testing.T) {
	_, err := ParseUser("Bob,abc")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
	if !errors.Is(err, ErrInvalidAge) {
		t.Errorf("expected ErrInvalidAge, got %v", err)
	}
}

func TestParseUserTrailingFields(t *testing.T) {
	_, err := ParseUser("Charlie,25,extra")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
	if !errors.Is(err, ErrInvalidFormat) {
		t.Errorf("expected ErrInvalidFormat, got %v", err)
	}
}

func TestParseUserParseErrorCarriesLine(t *testing.T) {
	_, err := ParseUser(",abc")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
	var pe *ParseError
	if errors.As(err, &pe) {
		if pe.Line != ",abc" {
			t.Errorf("Line = %q, want %q", pe.Line, ",abc")
		}
	}
}

func TestParseUserWhitespaceName(t *testing.T) {
	// スペースだけの名前は空とみなす
	_, err := ParseUser("  ,25")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
}

package main

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"encoding/base64"
	"encoding/json"
	"fmt"
	"os"
)

type SecretStore struct {
	key  [32]byte
	path string
	data map[string]string
}

func NewSecretStore() (*SecretStore, error) {
	s := &SecretStore{data: make(map[string]string)}
	if _, err := rand.Read(s.key[:]); err != nil {
		return nil, fmt.Errorf("generate key: %w", err)
	}
	return s, nil
}

func LoadFromFile(path string) (*SecretStore, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read file: %w", err)
	}
	decoded, err := base64.StdEncoding.DecodeString(string(data))
	if err != nil {
		return nil, fmt.Errorf("base64 decode: %w", err)
	}
	encKey := decoded[:32]
	nonce := decoded[32 : 32+12]
	cipherData := decoded[32+12:]

	block, err := aes.NewCipher(encKey)
	if err != nil {
		return nil, fmt.Errorf("aes: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return nil, fmt.Errorf("gcm: %w", err)
	}
	plaintext, err := gcm.Open(nil, nonce, cipherData, nil)
	if err != nil {
		return nil, fmt.Errorf("decrypt: %w", err)
	}

	var store SecretStore
	copy(store.key[:], encKey)
	store.path = path
	if err := json.Unmarshal(plaintext, &store.data); err != nil {
		return nil, fmt.Errorf("json: %w", err)
	}
	return &store, nil
}

func (s *SecretStore) SaveToFile(path string) error {
	data, err := json.Marshal(s.data)
	if err != nil {
		return fmt.Errorf("json marshal: %w", err)
	}
	block, err := aes.NewCipher(s.key[:])
	if err != nil {
		return fmt.Errorf("aes: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return fmt.Errorf("gcm: %w", err)
	}
	nonce := make([]byte, 12)
	if _, err := rand.Read(nonce); err != nil {
		return fmt.Errorf("nonce: %w", err)
	}
	ciphertext := gcm.Seal(nil, nonce, data, nil)
	encoded := base64.StdEncoding.EncodeToString(append(s.key[:], append(nonce, ciphertext...)...))
	return os.WriteFile(path, []byte(encoded), 0600)
}

func (s *SecretStore) Set(key, value string) {
	s.data[key] = value
}

func (s *SecretStore) Get(key string) string {
	return s.data[key]
}

func (s *SecretStore) Delete(key string) {
	delete(s.data, key)
}

var ErrSecretNotFound = fmt.Errorf("secret not found")

func GetEnvSecret(key string) (string, error) {
	val := os.Getenv(key)
	if val == "" {
		return "", ErrSecretNotFound
	}
	return val, nil
}

func Encrypt(key []byte, plaintext string) (string, error) {
	block, err := aes.NewCipher(key)
	if err != nil {
		return "", fmt.Errorf("aes: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return "", fmt.Errorf("gcm: %w", err)
	}
	nonce := make([]byte, 12)
	if _, err := rand.Read(nonce); err != nil {
		return "", fmt.Errorf("nonce: %w", err)
	}
	ciphertext := gcm.Seal(nil, nonce, []byte(plaintext), nil)
	return base64.StdEncoding.EncodeToString(append(nonce, ciphertext...)), nil
}

func Decrypt(key []byte, ciphertext string) (string, error) {
	decoded, err := base64.StdEncoding.DecodeString(ciphertext)
	if err != nil {
		return "", fmt.Errorf("base64: %w", err)
	}
	block, err := aes.NewCipher(key)
	if err != nil {
		return "", fmt.Errorf("aes: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return "", fmt.Errorf("gcm: %w", err)
	}
	nonce := decoded[:12]
	data := decoded[12:]
	plaintext, err := gcm.Open(nil, nonce, data, nil)
	if err != nil {
		return "", fmt.Errorf("decrypt: %w", err)
	}
	return string(plaintext), nil
}

func Clear(b []byte) {
	for i := range b {
		b[i] = 0
	}
}

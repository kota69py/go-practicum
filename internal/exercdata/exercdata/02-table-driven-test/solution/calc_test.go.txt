package calc

import "testing"

func TestMax(t *testing.T) {
	tests := []struct {
		name    string
		numbers []int
		want    int
		wantErr bool
	}{
		{name: "empty slice", numbers: []int{}, want: 0, wantErr: true},
		{name: "single element", numbers: []int{5}, want: 5},
		{name: "ascending", numbers: []int{1, 2, 3, 4, 5}, want: 5},
		{name: "descending", numbers: []int{9, 7, 5, 3, 1}, want: 9},
		{name: "unsorted", numbers: []int{3, 7, 1, 9, 4}, want: 9},
		{name: "negative numbers", numbers: []int{-5, -2, -8, -1}, want: -1},
		{name: "all same values", numbers: []int{3, 3, 3, 3}, want: 3},
		{name: "max at first", numbers: []int{100, 1, 2, 3}, want: 100},
		{name: "max at last", numbers: []int{1, 2, 3, 100}, want: 100},
		{name: "duplicate max", numbers: []int{5, 1, 5, 2}, want: 5},
	}
	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			got, err := Max(tt.numbers)
			if tt.wantErr {
				if err == nil {
					t.Error("expected error, got nil")
				}
				return
			}
			if err != nil {
				t.Fatalf("unexpected error: %v", err)
			}
			if got != tt.want {
				t.Errorf("Max() = %d, want %d", got, tt.want)
			}
		})
	}
}

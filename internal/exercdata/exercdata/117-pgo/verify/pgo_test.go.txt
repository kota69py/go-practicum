package main

import (
	"os"
	"path/filepath"
	"testing"
)

func TestGenerateCPUProfile(t *testing.T) {
	tmpDir := t.TempDir()
	profilePath := filepath.Join(tmpDir, "cpu.pprof")

	err := GenerateCPUProfile(profilePath)
	if err != nil {
		t.Fatalf("GenerateCPUProfile failed: %v", err)
	}

	if _, err := os.Stat(profilePath); os.IsNotExist(err) {
		t.Fatal("profile file was not created")
	}

	data, err := os.ReadFile(profilePath)
	if err != nil {
		t.Fatalf("failed to read profile: %v", err)
	}
	if len(data) == 0 {
		t.Fatal("profile file is empty")
	}
}

func TestFibonacciCorrectness(t *testing.T) {
	tests := []struct {
		n    int
		want int
	}{
		{0, 0}, {1, 1}, {2, 1}, {3, 2}, {4, 3},
		{5, 5}, {6, 8}, {10, 55}, {15, 610}, {20, 6765},
	}
	for _, tc := range tests {
		got := Fibonacci(tc.n)
		if got != tc.want {
			t.Errorf("Fibonacci(%d) = %d, want %d", tc.n, got, tc.want)
		}
	}
}

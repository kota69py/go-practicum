package main

import (
	"testing"
)

func TestCreateTask(t *testing.T) {
	repo := NewInMemoryTaskRepository()
	uc := NewTaskUseCase(repo)

	task, err := uc.CreateTask("Buy milk")
	if err != nil {
		t.Fatalf("CreateTask: %v", err)
	}
	if task.Title != "Buy milk" {
		t.Errorf("title: got %q, want %q", task.Title, "Buy milk")
	}
	if task.Done {
		t.Error("new task should not be done")
	}
	if task.ID == "" {
		t.Error("expected non-empty ID")
	}
	if task.CreatedAt.IsZero() {
		t.Error("expected non-zero CreatedAt")
	}
}

func TestEndToEnd(t *testing.T) {
	repo := NewInMemoryTaskRepository()
	uc := NewTaskUseCase(repo)

	t1, _ := uc.CreateTask("Task 1")
	t2, _ := uc.CreateTask("Task 2")
	uc.CompleteTask(t1.ID)

	tasks, _ := uc.ListTasks()
	if len(tasks) != 2 {
		t.Errorf("expected 2 tasks, got %d", len(tasks))
	}

	gotT1, _ := uc.GetTask(t1.ID)
	if !gotT1.Done {
		t.Error("expected t1 to be done")
	}

	gotT2, _ := uc.GetTask(t2.ID)
	if gotT2.Done {
		t.Error("expected t2 not to be done")
	}

	uc.DeleteTask(t1.ID)
	tasks, _ = uc.ListTasks()
	if len(tasks) != 1 {
		t.Errorf("expected 1 task after delete, got %d", len(tasks))
	}
}

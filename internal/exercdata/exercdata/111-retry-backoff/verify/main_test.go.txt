package main

import (
	"context"
	"errors"
	"testing"
	"time"
)

func TestDefaultRetryConfig(t *testing.T) {
	cfg := DefaultRetryConfig()
	if cfg.MaxAttempts != 3 {
		t.Errorf("MaxAttempts = %d, want 3", cfg.MaxAttempts)
	}
	if cfg.BaseDelay != 100*time.Millisecond {
		t.Errorf("BaseDelay = %v, want 100ms", cfg.BaseDelay)
	}
}

func TestRetrySuccess(t *testing.T) {
	attempts := 0
	err := Retry(context.Background(), DefaultRetryConfig(), func(ctx context.Context) error {
		attempts++
		return nil
	})
	if err != nil {
		t.Fatalf("Retry error: %v", err)
	}
	if attempts != 1 {
		t.Errorf("attempts = %d, want 1", attempts)
	}
}

func TestRetryEventuallySucceeds(t *testing.T) {
	attempts := 0
	err := Retry(context.Background(), DefaultRetryConfig(), func(ctx context.Context) error {
		attempts++
		if attempts < 2 {
			return errors.New("temporary error")
		}
		return nil
	})
	if err != nil {
		t.Fatalf("Retry error: %v", err)
	}
	if attempts != 2 {
		t.Errorf("attempts = %d, want 2", attempts)
	}
}

func TestRetryExhausted(t *testing.T) {
	attempts := 0
	err := Retry(context.Background(), DefaultRetryConfig(), func(ctx context.Context) error {
		attempts++
		return errors.New("persistent error")
	})
	if err == nil {
		t.Fatal("expected error")
	}
	var retryErr *RetryError
	if !errors.As(err, &retryErr) {
		t.Fatalf("expected RetryError, got %T", err)
	}
	if retryErr.Attempts != 3 {
		t.Errorf("Attempts = %d, want 3", retryErr.Attempts)
	}
	if len(retryErr.Errors) != 3 {
		t.Errorf("Errors count = %d, want 3", len(retryErr.Errors))
	}
}

func TestRetryContextCancel(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	cancel()

	err := Retry(ctx, DefaultRetryConfig(), func(ctx context.Context) error {
		return errors.New("error")
	})
	if err == nil {
		t.Fatal("expected error for cancelled context")
	}
}

func TestCalculateDelay(t *testing.T) {
	cfg := DefaultRetryConfig()
	delay := calculateDelay(0, cfg, 0)
	if delay != cfg.BaseDelay {
		t.Errorf("delay attempt 0 = %v, want %v", delay, cfg.BaseDelay)
	}
}

func TestFullJitter(t *testing.T) {
	delay := 1 * time.Second
	for i := 0; i < 10; i++ {
		j := FullJitter(delay)
		if j < 0 || j >= delay {
			t.Errorf("FullJitter = %v, want [0, %v)", j, delay)
		}
	}
}

func TestEqualJitter(t *testing.T) {
	delay := 1 * time.Second
	for i := 0; i < 10; i++ {
		j := EqualJitter(delay)
		half := delay / 2
		if j < half || j >= delay {
			t.Errorf("EqualJitter = %v, want [%v, %v)", j, half, delay)
		}
	}
}

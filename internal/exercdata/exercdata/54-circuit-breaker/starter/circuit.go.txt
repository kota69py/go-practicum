// Options は CircuitBreaker の設定です。
type Options struct {
	FailureThreshold  int
	Timeout           time.Duration
	HalfOpenMaxRetries int
}

// CircuitBreaker は障害分離のためのサーキットブレーカーです。
type CircuitBreaker struct {
	mu         sync.Mutex
	state      State
	failures   int
	lastError  time.Time
	opts       Options
}

// NewCircuitBreaker は CircuitBreaker を作成します。
func NewCircuitBreaker(opts Options) *CircuitBreaker {
	return &CircuitBreaker{
		state: StateClosed,
		opts:  opts,
	}
}

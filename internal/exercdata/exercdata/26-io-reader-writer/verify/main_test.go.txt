package main

import (
	"bytes"
	"io"
	"strings"
	"testing"
)

func TestROT13Reader(t *testing.T) {
	input := "Hello, World!"
	expected := "Uryyb, Jbeyq!"
	r := &ROT13Reader{r: strings.NewReader(input)}
	result, err := io.ReadAll(r)
	if err != nil {
		t.Fatalf("ReadAll error: %v", err)
	}
	if string(result) != expected {
		t.Errorf("got %q, want %q", string(result), expected)
	}
}

func TestROT13ReaderEmpty(t *testing.T) {
	r := &ROT13Reader{r: strings.NewReader("")}
	result, err := io.ReadAll(r)
	if err != nil {
		t.Fatalf("ReadAll error: %v", err)
	}
	if len(result) != 0 {
		t.Errorf("expected empty, got %q", string(result))
	}
}

func TestROT13RoundTrip(t *testing.T) {
	input := "Go is awesome!"
	r := &ROT13Reader{r: strings.NewReader(input)}
	encrypted, _ := io.ReadAll(r)

	r2 := &ROT13Reader{r: bytes.NewReader(encrypted)}
	decrypted, _ := io.ReadAll(r2)

	if string(decrypted) != input {
		t.Errorf("round trip: got %q, want %q", string(decrypted), input)
	}
}

func TestROT13Writer(t *testing.T) {
	var buf bytes.Buffer
	w := &ROT13Writer{w: &buf}
	n, err := w.Write([]byte("Hello"))
	if err != nil {
		t.Fatalf("Write error: %v", err)
	}
	if n != 5 {
		t.Errorf("wrote %d bytes, want 5", n)
	}
	if buf.String() != "Uryyb" {
		t.Errorf("got %q, want %q", buf.String(), "Uryyb")
	}
}

func TestROT13WriterEmpty(t *testing.T) {
	var buf bytes.Buffer
	w := &ROT13Writer{w: &buf}
	n, err := w.Write([]byte{})
	if err != nil {
		t.Fatalf("Write error: %v", err)
	}
	if n != 0 {
		t.Errorf("wrote %d bytes, want 0", n)
	}
}

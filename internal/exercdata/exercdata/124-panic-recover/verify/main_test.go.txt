package main

import (
	"errors"
	"net/http"
	"net/http/httptest"
	"testing"
)

func TestRecoveryMiddleware_Panic(t *testing.T) {
	panicHandler := http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		panic("something went wrong")
	})
	h := RecoveryMiddleware(panicHandler)
	srv := httptest.NewServer(h)
	defer srv.Close()

	resp, err := http.Get(srv.URL + "/panic")
	if err != nil {
		t.Fatalf("GET failed: %v", err)
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusInternalServerError {
		t.Errorf("status = %d, want %d", resp.StatusCode, http.StatusInternalServerError)
	}
}

func TestRecoveryMiddleware_OK(t *testing.T) {
	okHandler := http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.WriteHeader(http.StatusOK)
	})
	h := RecoveryMiddleware(okHandler)
	srv := httptest.NewServer(h)
	defer srv.Close()

	resp, err := http.Get(srv.URL + "/ok")
	if err != nil {
		t.Fatalf("GET failed: %v", err)
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		t.Errorf("status = %d, want %d", resp.StatusCode, http.StatusOK)
	}
}

func TestSafeCall_PanicString(t *testing.T) {
	err := SafeCall(func() error {
		panic("crash")
	})
	if err == nil {
		t.Fatal("SafeCall: expected error, got nil")
	}
	if err.Error() != "panic: crash" {
		t.Errorf("SafeCall = %q, want %q", err.Error(), "panic: crash")
	}
}

func TestSafeCall_PanicError(t *testing.T) {

	})

	}
}

func TestRepanic_Panics(t *testing.T) {
	defer func() {
		if rec := recover(); rec == nil {
			t.Fatal("Repanic: expected panic, got nil")

		panic("repanic test")
	})
}

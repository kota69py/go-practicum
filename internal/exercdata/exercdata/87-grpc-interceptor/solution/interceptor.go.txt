package main

import (
	"context"
	"fmt"
	"log"
	"time"

	"google.golang.org/grpc"
	"google.golang.org/grpc/codes"
	"google.golang.org/grpc/metadata"
	"google.golang.org/grpc/status"
)

func LoggingInterceptor(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
	start := time.Now()
	log.Printf("--> %s req=%+v", info.FullMethod, req)
	resp, err := handler(ctx, req)
	log.Printf("<-- %s dur=%s err=%v", info.FullMethod, time.Since(start), err)
	return resp, err
}

func RecoveryInterceptor(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (resp interface{}, err error) {
	defer func() {
		if r := recover(); r != nil {
			err = status.Errorf(codes.Internal, "panic: %v", r)
		}
	}()
	return handler(ctx, req)
}

func AuthInterceptor(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
	md, ok := metadata.FromIncomingContext(ctx)
	if !ok {
		return nil, status.Error(codes.Unauthenticated, "missing metadata")
	}
	tokens := md["authorization"]
	if len(tokens) == 0 || tokens[0] != "valid-token" {
		return nil, status.Error(codes.Unauthenticated, "invalid token")
	}
	ctx = context.WithValue(ctx, ctxKeyUser{}, "authenticated-user")
	return handler(ctx, req)
}

var rateLimitCh = make(chan struct{}, 1)

func init() {
	rateLimitCh <- struct{}{}
}

func RateLimitInterceptor(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
	select {
	case <-rateLimitCh:
		defer func() { rateLimitCh <- struct{}{} }()
		return handler(ctx, req)
	default:
		return nil, status.Error(codes.ResourceExhausted, "rate limit exceeded")
	}
}

type ctxKeyUser struct{}

func UserFromContext(ctx context.Context) (string, bool) {
	u, ok := ctx.Value(ctxKeyUser{}).(string)
	return u, ok
}

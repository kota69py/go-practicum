func TestSendEvent(t *testing.T) {
	var buf strings.Builder
	ev := Event{Data: "test", ID: "1"}

	}
}

func TestSSEHandler_headers(t *testing.T) {
	handler := http.HandlerFunc(SSEHandler)
	req := httptest.NewRequest(http.MethodGet, "/", nil)
	rec := httptest.NewRecorder()
	handler.ServeHTTP(rec, req)

	if rec.Code != http.StatusOK {
		t.Errorf("status = %d, want %d", rec.Code, http.StatusOK)
	}
	if ct := rec.Header().Get("Content-Type"); ct != "text/event-stream" {
		t.Errorf("Content-Type = %q, want %q", ct, "text/event-stream")
	}
	if cc := rec.Header().Get("Cache-Control"); cc != "no-cache" {
		t.Errorf("Cache-Control = %q, want %q", cc, "no-cache")
	}
}

func TestSSEHandler_streamsEvents(t *testing.T) {
	handler := http.HandlerFunc(SSEHandler)
	req := httptest.NewRequest(http.MethodGet, "/", nil)
	rec := httptest.NewRecorder()
	handler.ServeHTTP(rec, req)

	body := rec.Body.String()
	lines := strings.Split(strings.TrimSpace(body), "\n")

	gotData := false
	gotEventStart := false
	gotEventEnd := false
	for _, line := range lines {
		if strings.HasPrefix(line, "data:") {
			gotData = true
		}
		if line == "event: start" {
			gotEventStart = true
		}
		if line == "event: end" {
			gotEventEnd = true
		}
	}
	if !gotData {
		t.Error("expected at least one 'data:' line")
	}
	if !gotEventStart {
		t.Error("expected 'event: start'")
	}
	if !gotEventEnd {
		t.Error("expected 'event: end'")
	}
}

func TestSSEHandler_eventsCount(t *testing.T) {
	handler := http.HandlerFunc(SSEHandler)
	req := httptest.NewRequest(http.MethodGet, "/", nil)
	rec := httptest.NewRecorder()
	handler.ServeHTTP(rec, req)

	// Count events by looking for blank-line separators
	scanner := bufio.NewScanner(strings.NewReader(rec.Body.String()))
	eventCount := 0
	for scanner.Scan() {
		if scanner.Text() == "" {
			eventCount++
		}
	}
	// Expect 5 events (each ends with blank line)
	if eventCount != 5 {
		t.Errorf("got %d events, want 5", eventCount)
	}
}

func TestSSEHandler_flusherRequired(t *testing.T) {
	// A ResponseWriter that does NOT implement http.Flusher
	handler := http.HandlerFunc(SSEHandler)
	req := httptest.NewRequest(http.MethodGet, "/", nil)
	rec := &nonFlushResponseWriter{rec: httptest.NewRecorder()}
	handler.ServeHTTP(rec, req)

	if rec.rec.Code != http.StatusNotImplemented {
		t.Errorf("status = %d, want %d", rec.rec.Code, http.StatusNotImplemented)
	}
}

// nonFlushResponseWriter wraps httptest.ResponseRecorder without Flush.
type nonFlushResponseWriter struct {
	rec *httptest.ResponseRecorder
}

func (w *nonFlushResponseWriter) Header() http.Header {
	return w.rec.Header()
}

func (w *nonFlushResponseWriter) Write(b []byte) (int, error) {
	return w.rec.Write(b)
}

func (w *nonFlushResponseWriter) WriteHeader(code int) {
	w.rec.WriteHeader(code)
}

func TestSSEHandler_contextCancel(t *testing.T) {
	handler := http.HandlerFunc(SSEHandler)
	req := httptest.NewRequest(http.MethodGet, "/", nil)
	cancelCtx, cancel := context.WithCancel(req.Context())
	req = req.WithContext(cancelCtx)
	rec := httptest.NewRecorder()

	// Cancel immediately to test graceful shutdown
	cancel()

	handler.ServeHTTP(rec, req)
	// Should not panic, should return quickly
}

func TestRealSSEOverHTTP(t *testing.T) {
	handler := http.HandlerFunc(SSEHandler)
	ts := httptest.NewServer(handler)
	defer ts.Close()

	resp, err := http.Get(ts.URL)
	if err != nil {
		t.Fatal(err)
	}
	defer resp.Body.Close()

	if ct := resp.Header.Get("Content-Type"); ct != "text/event-stream" {
		t.Errorf("Content-Type = %q, want %q", ct, "text/event-stream")
	}

	// Read at least the first event
	body, err := io.ReadAll(resp.Body)
	if err != nil {
		t.Fatal(err)
	}
	if !strings.Contains(string(body), "data:") {
		t.Error("response body contains no data: lines")
	}
}

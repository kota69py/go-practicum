package main

import (
	"context"
	"strings"
	"testing"
	"time"
)

func TestSearchReturnsResult(t *testing.T) {
	ctx := context.Background()
	result, err := Search(ctx, "golang")
	if err != nil {
		t.Fatalf("unexpected error: %v", err)
	}
	if !strings.Contains(result, "golang") {
		t.Errorf("result %q does not contain query %q", result, "golang")
	}
}

func TestSearchReturnsFastest(t *testing.T) {
	// google (10ms) が bing (20ms) や yahoo (30ms) より先に返るはず
	start := time.Now()
	ctx := context.Background()
	result, err := Search(ctx, "go")
	elapsed := time.Since(start)
	if err != nil {
		t.Fatalf("unexpected error: %v", err)
	}
	if !strings.Contains(result, "google") {
		t.Logf("result came from %q (not necessarily wrong, but expected google)", result)
	}
	// 最速の google は 10ms、でも待ち時間は 10-20ms のはず
	if elapsed > 50*time.Millisecond {
		t.Errorf("took %v, expected ~10ms (fastest)", elapsed)
	}
}

func TestSearchContextCancel(t *testing.T) {
	ctx, cancel := context.WithCancel(context.Background())
	cancel() // 即座にキャンセル

	_, err := Search(ctx, "golang")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
	if err != context.Canceled {
		t.Errorf("expected Canceled, got %v", err)
	}
}

func TestSearchContextTimeout(t *testing.T) {
	ctx, cancel := context.WithTimeout(context.Background(), 1*time.Millisecond)
	defer cancel()

	_, err := Search(ctx, "golang")
	if err == nil {
		t.Fatal("expected error, got nil")
	}
	if err != context.DeadlineExceeded {
		t.Errorf("expected DeadlineExceeded, got %v", err)
	}
}

func TestSearchMultipleCalls(t *testing.T) {
	ctx := context.Background()
	for i := 0; i < 3; i++ {
		result, err := Search(ctx, "go")
		if err != nil {
			t.Fatalf("call %d: unexpected error: %v", i, err)
		}
		if result == "" {
			t.Errorf("call %d: empty result", i)
		}
	}
}

package main

import (
	"context"
	"fmt"
	"time"
)

var searchDelays = map[string]time.Duration{
	"google": 10 * time.Millisecond,
	"bing":   20 * time.Millisecond,
	"yahoo":  30 * time.Millisecond,
}

func searchSource(ctx context.Context, name, query string) (string, error) {
	delay := searchDelays[name]
	select {
	case <-time.After(delay):
		return fmt.Sprintf("%s result for %q", name, query), nil
	case <-ctx.Done():
		return "", ctx.Err()
	}
}

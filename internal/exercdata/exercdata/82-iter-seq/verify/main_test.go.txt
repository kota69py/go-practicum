package main

import (
	"slices"
	"testing"
)

func makeTree() *Node {
	return &Node{Value: 1,
		Left: &Node{Value: 2,
			Left:  &Node{Value: 4},
			Right: &Node{Value: 5}},
		Right: &Node{Value: 3,
			Left: &Node{Value: 6}}}
}

func TestInorder(t *testing.T) {
	tree := makeTree()
	got := slices.Collect(tree.Inorder())
	want := []int{4, 2, 5, 1, 6, 3}
	if !slices.Equal(got, want) {
		t.Errorf("inorder: got %v, want %v", got, want)
	}
}

func TestBFS(t *testing.T) {
	tree := makeTree()
	got := slices.Collect(tree.BFS())
	want := []int{1, 2, 3, 4, 5, 6}
	if !slices.Equal(got, want) {
		t.Errorf("bfs: got %v, want %v", got, want)
	}
}

func TestPaths(t *testing.T) {
	tree := makeTree()
	got := slices.Collect(tree.Paths())
	if len(got) != 3 {
		t.Fatalf("expected 3 paths, got %d: %v", len(got), got)
	}
}

func TestInorderEmpty(t *testing.T) {
	var tree *Node
	got := slices.Collect(tree.Inorder())
	if len(got) != 0 {
		t.Errorf("empty tree: got %v, want []", got)
	}
}

func TestBFSEmpty(t *testing.T) {
	var tree *Node
	got := slices.Collect(tree.BFS())
	if len(got) != 0 {
		t.Errorf("empty tree: got %v, want []", got)
	}
}

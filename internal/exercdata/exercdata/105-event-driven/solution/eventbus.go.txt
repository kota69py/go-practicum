package main

import (
	"fmt"
	"sync"
)

type Event interface {
	Type() string
	Data() interface{}
}

type BaseEvent struct {
	eventType string
	data      interface{}
}

func (e BaseEvent) Type() string      { return e.eventType }
func (e BaseEvent) Data() interface{} { return e.data }

type EventHandler interface {
	Handle(event Event) error
}

type EventHandlerFunc func(event Event) error

func (f EventHandlerFunc) Handle(event Event) error {
	return f(event)
}

type EventBus struct {
	handlers map[string][]EventHandler
	mu       sync.RWMutex
	wg       sync.WaitGroup
	errCh    chan error
}

func NewEventBus() *EventBus {
	return &EventBus{
		handlers: make(map[string][]EventHandler),
		errCh:    make(chan error, 100),
	}
}

func (eb *EventBus) Subscribe(eventType string, handler EventHandler) {
	eb.mu.Lock()
	defer eb.mu.Unlock()
	eb.handlers[eventType] = append(eb.handlers[eventType], handler)
}

func (eb *EventBus) Unsubscribe(eventType string, handler EventHandler) {
	eb.mu.Lock()
	defer eb.mu.Unlock()
	handlers := eb.handlers[eventType]
	for i, h := range handlers {
		if fmt.Sprintf("%p", h) == fmt.Sprintf("%p", handler) {
			eb.handlers[eventType] = append(handlers[:i], handlers[i+1:]...)
			return
		}
	}
}

func (eb *EventBus) Publish(event Event) {
	eb.mu.RLock()
	handlers := eb.handlers[event.Type()]
	wildcardHandlers := eb.handlers["*"]
	eb.mu.RUnlock()

	for _, h := range handlers {
		h.Handle(event)
	}
	for _, h := range wildcardHandlers {
		h.Handle(event)
	}
}

func (eb *EventBus) PublishAsync(event Event) {
	eb.mu.RLock()
	handlers := eb.handlers[event.Type()]
	wildcardHandlers := eb.handlers["*"]
	eb.mu.RUnlock()

	allHandlers := append(handlers, wildcardHandlers...)
	for _, h := range allHandlers {
		eb.wg.Add(1)
		go func(handler EventHandler) {
			defer eb.wg.Done()
			if err := handler.Handle(event); err != nil {
				select {
				case eb.errCh <- err:
				default:
				}
			}
		}(h)
	}
}

func (eb *EventBus) Wait() {
	eb.wg.Wait()
}

func (eb *EventBus) Errors() <-chan error {
	return eb.errCh
}

func (eb *EventBus) Close() {
	close(eb.errCh)
}

type UserCreatedEvent struct {
	BaseEvent
}

func NewUserCreatedEvent(userID, name string) UserCreatedEvent {
	return UserCreatedEvent{
		BaseEvent: BaseEvent{
			eventType: "user.created",
			data:      map[string]string{"user_id": userID, "name": name},
		},
	}
}

type OrderPlacedEvent struct {
	BaseEvent
}

func NewOrderPlacedEvent(orderID, userID string, total float64) OrderPlacedEvent {
	return OrderPlacedEvent{
		BaseEvent: BaseEvent{
			eventType: "order.placed",
			data:      map[string]interface{}{"order_id": orderID, "user_id": userID, "total": total},
		},
	}
}

type LoggingHandler struct{}

func (h LoggingHandler) Handle(event Event) error {
	fmt.Printf("[EVENT] %s: %v\n", event.Type(), event.Data())
	return nil
}

type EmailHandler struct{}

func (h EmailHandler) Handle(event Event) error {
	data := event.Data().(map[string]string)
	fmt.Printf("[EMAIL] Sending welcome email to user %s (%s)\n", data["user_id"], data["name"])
	return nil
}

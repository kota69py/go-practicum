package main

import (
	"os"
	"path/filepath"
	"strings"
	"testing"
)

func TestParseFile(t *testing.T) {
	dir := t.TempDir()
	src := `package test
type Person struct {
	Name string
	Age  int
}`
	f := filepath.Join(dir, "person.go")
	if err := os.WriteFile(f, []byte(src), 0644); err != nil {
		t.Fatal(err)
	}
	structs, err := ParseFile(f)
	if err != nil {
		t.Fatalf("ParseFile error: %v", err)
	}
	if len(structs) != 1 {
		t.Fatalf("got %d structs, want 1", len(structs))
	}
	if structs[0].Name != "Person" {
		t.Errorf("Name = %q, want %q", structs[0].Name, "Person")
	}
	if len(structs[0].Fields) != 2 {
		t.Fatalf("got %d fields, want 2", len(structs[0].Fields))
	}
}

func TestParseFileWithTags(t *testing.T) {
	dir := t.TempDir()
	src := `package test
type Config struct {
	Host string ` + "`" + `json:"host" validate:"required"` + "`" + `
	Port int    ` + "`" + `json:"port"` + "`" + `
}`
	f := filepath.Join(dir, "config.go")
	if err := os.WriteFile(f, []byte(src), 0644); err != nil {
		t.Fatal(err)
	}
	structs, err := ParseFile(f)
	if err != nil {
		t.Fatalf("ParseFile error: %v", err)
	}
	if len(structs) != 1 {
		t.Fatal("expected 1 struct")
	}
	if len(structs[0].Fields) != 2 {
		t.Fatalf("expected 2 fields, got %d", len(structs[0].Fields))
	}
}

func TestGenerateStringer(t *testing.T) {
	info := StructInfo{
		Name: "Person",
		Fields: []FieldInfo{
			{Name: "Name", Type: "string"},
			{Name: "Age", Type: "int"},
		},
	}
	code, err := GenerateStringer(info)
	if err != nil {
		t.Fatalf("GenerateStringer error: %v", err)
	}
	if !strings.Contains(code, "func (s *Person) String() string") {
		t.Errorf("generated code missing String method:\n%s", code)
	}
	if !strings.Contains(code, "s.Name") && !strings.Contains(code, "s.Age") {
		t.Errorf("generated code missing field references:\n%s", code)
	}
}

func TestGenerateAll(t *testing.T) {
	info := StructInfo{Name: "Test", Fields: []FieldInfo{{Name: "X", Type: "string"}}}
	code, err := GenerateAll(info, GenerateOptions{Stringer: true})
	if err != nil {
		t.Fatalf("GenerateAll error: %v", err)
	}
	if _, ok := code["String"]; !ok {
		t.Error("expected String method in generated code")
	}
}

func TestWriteGeneratedFile(t *testing.T) {
	dir := t.TempDir()
	f := filepath.Join(dir, "output.go")
	if err := WriteGeneratedFile(f, "package main"); err != nil {
		t.Fatalf("WriteGeneratedFile error: %v", err)
	}
	data, err := os.ReadFile(f)
	if err != nil {
		t.Fatal(err)
	}
	content := string(data)
	if !strings.Contains(content, "Code generated by") {
		t.Errorf("missing generated header:\n%s", content)
	}
}

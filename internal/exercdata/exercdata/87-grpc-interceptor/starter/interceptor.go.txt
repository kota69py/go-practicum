// contextKey は検証済みユーザー情報を context に保存するためのキーです。
type ctxKeyUser struct{}

// UserFromContext は context からユーザー情報を取得します。
func UserFromContext(ctx context.Context) (string, bool) {
	u, ok := ctx.Value(ctxKeyUser{}).(string)
	return u, ok
}

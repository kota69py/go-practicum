package main

import "fmt"

type PaymentGateway interface {
	Charge(amount int) (string, error)
	Refund(transactionID string) error
}

type PurchaseService struct {
	gateway PaymentGateway
}

func NewPurchaseService(g PaymentGateway) *PurchaseService {
	return &PurchaseService{gateway: g}
}

func (s *PurchaseService) Purchase(amount int) (string, error) {
	txID, err := s.gateway.Charge(amount)
	if err != nil {
		return "", fmt.Errorf("purchase failed: %w", err)
	}
	return txID, nil
}

package main

import (
	"bytes"
	"context"
	"encoding/json"
	"log/slog"
	"sync"
)

type attrEntry struct {
	attr   slog.Attr
	groups []string
}

type MapHandler struct {
	mu      sync.Mutex
	buf     *bytes.Buffer
	entries []attrEntry
	groups  []string
}

func NewMapHandler(buf *bytes.Buffer) *MapHandler {
	return &MapHandler{buf: buf}
}

func (h *MapHandler) Enabled(ctx context.Context, level slog.Level) bool {
	return true
}

func (h *MapHandler) Handle(ctx context.Context, r slog.Record) error {
	h.mu.Lock()
	defer h.mu.Unlock()

	m := make(map[string]any)
	if !r.Time.IsZero() {
		m[slog.TimeKey] = r.Time
	}
	m[slog.LevelKey] = r.Level
	m[slog.MessageKey] = r.Message
	for _, e := range h.entries {
		addAttr(m, e.attr, e.groups)
	}
	r.Attrs(func(a slog.Attr) bool {
		addAttr(m, a, h.groups)
		return true
	})

	data, err := json.Marshal(m)
	if err != nil {
		return err
	}
	h.buf.Write(data)
	h.buf.WriteByte('\n')
	return nil
}

func (h *MapHandler) WithAttrs(attrs []slog.Attr) slog.Handler {
	c := h.clone()
	for _, a := range attrs {
		entry := attrEntry{
			attr:   a,
			groups: append([]string(nil), c.groups...),
		}
		c.entries = append(c.entries, entry)
	}
	return c
}

func (h *MapHandler) WithGroup(name string) slog.Handler {
	c := h.clone()
	c.groups = append(c.groups, name)
	return c
}

func (h *MapHandler) clone() *MapHandler {
	return &MapHandler{
		buf:     h.buf,
		entries: append([]attrEntry(nil), h.entries...),
		groups:  append([]string(nil), h.groups...),
	}
}

func addAttr(m map[string]any, a slog.Attr, groups []string) {
	v := a.Value.Resolve()

	target := m
	for _, g := range groups {
		nested, ok := target[g].(map[string]any)
		if !ok {
			nested = make(map[string]any)
			target[g] = nested
		}
		target = nested
	}

	if v.Kind() == slog.KindGroup {
		if a.Key != "" {
			nested := make(map[string]any)
			target[a.Key] = nested
			for _, attr := range v.Group() {
				addAttr(nested, attr, nil)
			}
		} else {
			for _, attr := range v.Group() {
				addAttr(target, attr, nil)
			}
		}
		return
	}

	if a.Key == "" {
		return
	}
	target[a.Key] = v.Any()
}

package main

import "fmt"

func HexEncode(src []byte) string {
	return fmt.Sprintf("%x", src)
}

func HexDecode(src string) ([]byte, error) {
	if len(src)%2 != 0 {
		return nil, fmt.Errorf("odd length: %d", len(src))
	}
	dst := make([]byte, len(src)/2)
	for i := 0; i < len(src); i += 2 {
		hi, ok := hexVal(src[i])
		if !ok {
			return nil, fmt.Errorf("invalid hex char: %c", src[i])
		}
		lo, ok := hexVal(src[i+1])
		if !ok {
			return nil, fmt.Errorf("invalid hex char: %c", src[i+1])
		}
		dst[i/2] = hi<<4 | lo
	}
	return dst, nil
}

func hexVal(c byte) (byte, bool) {
	switch {
	case '0' <= c && c <= '9':
		return c - '0', true
	case 'a' <= c && c <= 'f':
		return c - 'a' + 10, true
	case 'A' <= c && c <= 'F':
		return c - 'A' + 10, true
	default:
		return 0, false
	}
}

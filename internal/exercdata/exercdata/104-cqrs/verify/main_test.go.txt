package main

import (
	"testing"
)

func TestCreateOrder(t *testing.T) {
	store := NewInMemoryOrderStore()
	cmdSvc := NewOrderCommandService(store)

	id, err := cmdSvc.CreateOrder(CreateOrderCommand{
		Customer: "Alice",
		Items: []OrderItem{
			{ProductID: "p1", Quantity: 2, Price: 10.0},
			{ProductID: "p2", Quantity: 1, Price: 5.0},
		},
	})
	if err != nil {
		t.Fatalf("CreateOrder: %v", err)
	}
	if id == "" {
		t.Fatal("expected non-empty ID")
	}
}

func TestQueryOrder(t *testing.T) {
	store := NewInMemoryOrderStore()
	cmdSvc := NewOrderCommandService(store)
	querySvc := NewOrderQueryService(store)

	id, _ := cmdSvc.CreateOrder(CreateOrderCommand{
		Customer: "Bob",
		Items:    []OrderItem{{ProductID: "p1", Quantity: 1, Price: 100.0}},
	})

	detail, err := querySvc.GetOrder(id)
	if err != nil {
		t.Fatalf("GetOrder: %v", err)
	}
	if detail.Customer != "Bob" {
		t.Errorf("customer: got %q, want %q", detail.Customer, "Bob")
	}
	if detail.Total != 100.0 {
		t.Errorf("total: got %f, want %f", detail.Total, 100.0)
	}
	if detail.Status != "pending" {
		t.Errorf("status: got %q, want %q", detail.Status, "pending")
	}
	if len(detail.Items) != 1 {
		t.Errorf("items: got %d, want 1", len(detail.Items))
	}
}

func TestListOrdersByStatus(t *testing.T) {
	store := NewInMemoryOrderStore()
	cmdSvc := NewOrderCommandService(store)
	querySvc := NewOrderQueryService(store)

	cmdSvc.CreateOrder(CreateOrderCommand{Customer: "A", Items: []OrderItem{{ProductID: "p1", Quantity: 1, Price: 10.0}}})
	id2, _ := cmdSvc.CreateOrder(CreateOrderCommand{Customer: "B", Items: []OrderItem{{ProductID: "p2", Quantity: 1, Price: 20.0}}})
	cmdSvc.ShipOrder(ShipOrderCommand{OrderID: id2})

	pendingOrders, _ := querySvc.ListOrdersByStatus("pending")
	shippedOrders, _ := querySvc.ListOrdersByStatus("shipped")

	if len(pendingOrders) != 1 {
		t.Errorf("pending: got %d, want 1", len(pendingOrders))
	}
	if len(shippedOrders) != 1 {
		t.Errorf("shipped: got %d, want 1", len(shippedOrders))
	}
}

func TestCommandQuerySeparation(t *testing.T) {
	store := NewInMemoryOrderStore()
	cmdSvc := NewOrderCommandService(store)
	querySvc := NewOrderQueryService(store)

	_, err := querySvc.GetOrder("999")
	if err == nil {
		t.Error("expected error for non-existent order")
	}

	_, errCmd := cmdSvc.ShipOrder(ShipOrderCommand{OrderID: "999"})
	if errCmd == nil {
		t.Error("expected error for shipping non-existent order")
	}
}

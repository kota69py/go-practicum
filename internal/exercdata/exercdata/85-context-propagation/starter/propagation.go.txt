// contextKey は context に TraceContext を保存するためのキーです。
type ctxKey struct{}

// ContextWithTrace は context に TraceContext を格納します。
func ContextWithTrace(ctx context.Context, tc TraceContext) context.Context {
	return context.WithValue(ctx, ctxKey{}, tc)
}

// TraceFromContext は context から TraceContext を取得します。
func TraceFromContext(ctx context.Context) (TraceContext, bool) {
	tc, ok := ctx.Value(ctxKey{}).(TraceContext)
	return tc, ok
}

// GetTraceID は context から TraceID を取得するヘルパーです（ハンドラ内で使用）。
func GetTraceID(ctx context.Context) string {
	tc, ok := TraceFromContext(ctx)
	if !ok {
		return ""
	}
	return tc.TraceID
}

// GetSpanID は context から SpanID を取得するヘルパーです。
func GetSpanID(ctx context.Context) string {
	tc, ok := TraceFromContext(ctx)
	if !ok {
		return ""
	}
	return tc.SpanID
}

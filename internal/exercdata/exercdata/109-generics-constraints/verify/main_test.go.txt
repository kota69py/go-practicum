package main

import (
	"testing"
)

func TestMin(t *testing.T) {
	if got := Min(3, 5); got != 3 {
		t.Errorf("Min(3,5) = %d, want 3", got)
	}
	if got := Min(5, 3); got != 3 {
		t.Errorf("Min(5,3) = %d, want 3", got)
	}
}

func TestMinFloat(t *testing.T) {
	if got := Min(3.14, 2.71); got != 2.71 {
		t.Errorf("Min(3.14,2.71) = %f, want 2.71", got)
	}
}

func TestMax(t *testing.T) {
	if got := Max(3, 5); got != 5 {
		t.Errorf("Max(3,5) = %d, want 5", got)
	}
}

func TestSum(t *testing.T) {
	if got := Sum([]int{1, 2, 3, 4, 5}); got != 15 {
		t.Errorf("Sum = %d, want 15", got)
	}
}

func TestSumFloat(t *testing.T) {
	if got := Sum([]float64{1.5, 2.5}); got != 4.0 {
		t.Errorf("Sum = %f, want 4.0", got)
	}
}

type MyInt int
type MyString string

func (m MyString) String() string { return string(m) }

func TestMinCustomType(t *testing.T) {
	a, b := MyInt(10), MyInt(20)
	if got := Min(a, b); got != MyInt(10) {
		t.Errorf("Min(10,20) = %d, want 10", got)
	}
}

func TestFormat(t *testing.T) {
	got := Format(MyString("hello"))
	if got != "hello" {
		t.Errorf("Format = %q, want %q", got, "hello")
	}
}

func TestOrderedSlice(t *testing.T) {
	var s OrderedSlice[int]
	s.Add(3)
	s.Add(1)
	s.Add(2)

	if !s.Contains(2) {
		t.Error("Contains(2) should be true")
	}
	if s.Contains(4) {
		t.Error("Contains(4) should be false")
	}
}

func TestOrderedSliceString(t *testing.T) {
	var s OrderedSlice[string]
	s.Add("c")
	s.Add("a")
	s.Add("b")

	if !s.Contains("a") {
		t.Error("Contains(a) should be true")
	}
	if s.Contains("d") {
		t.Error("Contains(d) should be false")
	}
}

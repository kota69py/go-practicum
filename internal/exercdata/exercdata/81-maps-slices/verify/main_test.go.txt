package main

import (
	"slices"
	"testing"
)

func TestSortStrings(t *testing.T) {
	s := []string{"c", "a", "b"}
	got := SortStrings(s)
	want := []string{"a", "b", "c"}
	if !slices.Equal(got, want) {
		t.Errorf("got %v, want %v", got, want)
	}
}

func TestSortDesc(t *testing.T) {
	n := []int{3, 1, 4, 1, 5}
	got := SortDesc(n)
	want := []int{5, 4, 3, 1, 1}
	if !slices.Equal(got, want) {
		t.Errorf("got %v, want %v", got, want)
	}
}

func TestFindIndex(t *testing.T) {
	s := []int{10, 20, 30, 20}
	if got := FindIndex(s, 20); got != 1 {
		t.Errorf("got %d, want 1", got)
	}
	if got := FindIndex(s, 99); got != -1 {
		t.Errorf("got %d, want -1", got)
	}
}

func TestFindIndexString(t *testing.T) {
	s := []string{"go", "rust", "zig"}
	if got := FindIndex(s, "rust"); got != 1 {
		t.Errorf("got %d, want 1", got)
	}
}

func TestCompactCopy(t *testing.T) {
	in := []int{1, 1, 2, 3, 3, 3, 4}
	got := CompactCopy(in)
	want := []int{1, 2, 3, 4}
	if !slices.Equal(got, want) {
		t.Errorf("got %v, want %v", got, want)
	}
	if len(in) != 7 {
		t.Error("CompactCopy should not modify original")
	}
}

func TestMergeUnique(t *testing.T) {
	got := MergeUnique([]int{1, 2, 3}, []int{3, 4, 5})
	if len(got) != 5 {
		t.Errorf("got len %d, want 5; got %v", len(got), got)
	}
}

func TestMaxValueKey(t *testing.T) {
	m := map[string]int{"a": 10, "b": 30, "c": 20}
	if got := MaxValueKey(m); got != "b" {
		t.Errorf("got %q, want %q", got, "b")
	}
}

func TestMaxValueKeyEmpty(t *testing.T) {
	if got := MaxValueKey(nil); got != "" {
		t.Errorf("got %q, want empty", got)
	}
}

func TestFirstNonZero(t *testing.T) {
	tests := []struct {
		in   []int
		want int
	}{
		{[]int{0, 0, 5, 0}, 5},
		{[]int{0, 0, 0}, 0},
		{[]int{42}, 42},
		{nil, 0},
	}
	for _, tt := range tests {
		got := FirstNonZero(tt.in...)
		if got != tt.want {
			t.Errorf("FirstNonZero(%v) = %d, want %d", tt.in, got, tt.want)
		}
	}
}

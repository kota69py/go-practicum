package main

import "fmt"

type User struct {
	Name string
	Age  int
}

// ParseError は入力行のパースに失敗したことを表すカスタムエラー型です。
type ParseError struct {
	Line    string
	Message string
}

func (e *ParseError) Error() string {
	return fmt.Sprintf("line %q: %s", e.Line, e.Message)
}

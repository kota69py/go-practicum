package main

import "context"

// WorkflowState represents the current state of a workflow
type WorkflowState string

// WorkflowStatus represents the overall status
type WorkflowStatus string

const (
	StatusPending    WorkflowStatus = "pending"
	StatusRunning    WorkflowStatus = "running"
	StatusCompleted  WorkflowStatus = "completed"
	StatusFailed     WorkflowStatus = "failed"
	StatusRolledBack WorkflowStatus = "rolled_back"
)

// TaskResult holds the result of a single task execution
type TaskResult struct {
	TaskName string
	Error    error
	Output   any
}

// Task defines a single unit of work in the workflow
type Task struct {
	Name    string
	Execute func(ctx context.Context) (any, error)
	// Rollback undoes the task. Called when the workflow fails after this task.
	Rollback func(ctx context.Context, output any) error
}

// Workflow is a directed acyclic graph of tasks.
type Workflow struct {
	Name         string
	Tasks        map[string]*Task
	Dependencies map[string][]string
}

// Execution tracks a single run of a workflow
type Execution struct {
	ID        string
	Workflow  *Workflow
	Status    WorkflowStatus
	State     map[string]any
	Completed []string
	Failed    []string
}

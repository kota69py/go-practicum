package main

import (
	"bufio"
	"fmt"
	"os"
	"os/exec"
	"path/filepath"
	"strings"
)

func InitWorkspace(dir string) error {
	info, err := os.Stat(dir)
	if err != nil {
		return fmt.Errorf("directory error: %w", err)
	}
	if !info.IsDir() {
		return fmt.Errorf("%s is not a directory", dir)
	}

	content := "go 1.23\n"
	workPath := filepath.Join(dir, "go.work")
	if err := os.WriteFile(workPath, []byte(content), 0644); err != nil {
		return fmt.Errorf("failed to write go.work: %w", err)
	}
	return nil
}

func UseModule(workDir, moduleDir string) error {
	workPath := filepath.Join(workDir, "go.work")
	f, err := os.OpenFile(workPath, os.O_RDWR, 0644)
	if err != nil {
		return fmt.Errorf("failed to open go.work: %w", err)
	}
	defer f.Close()

	var lines []string
	scanner := bufio.NewScanner(f)
	hasUseSection := false
	inUseBlock := false
	for scanner.Scan() {
		line := scanner.Text()
		trimmed := strings.TrimSpace(line)
		if trimmed == "use (" {
			inUseBlock = true
			hasUseSection = true
		}
		if inUseBlock && trimmed == ")" {
			inUseBlock = false
		}
		lines = append(lines, line)
	}
	if err := scanner.Err(); err != nil {
		return fmt.Errorf("failed to read go.work: %w", err)
	}

	if !hasUseSection {
		lines = append(lines, "", "use (", "\t./"+moduleDir, ")")
	} else {
		for i, line := range lines {
			trimmed := strings.TrimSpace(line)
			if trimmed == ")" && i > 0 {
				before := strings.TrimSpace(lines[i-1])
				if before == "" || strings.HasPrefix(before, "./") {
					lines = append(lines[:i], append([]string{"\t./" + moduleDir}, lines[i:]...)...)
					break
				}
			}
		}
	}

	output := strings.Join(lines, "\n")
	if err := os.WriteFile(workPath, []byte(output), 0644); err != nil {
		return fmt.Errorf("failed to write go.work: %w", err)
	}
	return nil
}

func BuildAll(workDir string) error {
	workPath := filepath.Join(workDir, "go.work")
	f, err := os.Open(workPath)
	if err != nil {
		return fmt.Errorf("failed to open go.work: %w", err)
	}
	defer f.Close()

	var modules []string
	scanner := bufio.NewScanner(f)
	inUseBlock := false
	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())
		if line == "use (" {
			inUseBlock = true
			continue
		}
		if inUseBlock && line == ")" {
			inUseBlock = false
			continue
		}
		if inUseBlock {
			modPath := strings.TrimPrefix(line, "./")
			modules = append(modules, modPath)
		}
	}
	if err := scanner.Err(); err != nil {
		return fmt.Errorf("failed to read go.work: %w", err)
	}

	if len(modules) == 0 {
		return fmt.Errorf("no modules found in workspace")
	}

	for _, mod := range modules {
		modDir := filepath.Join(workDir, mod)
		cmd := exec.Command("go", "build", "./...")
		cmd.Dir = modDir
		cmd.Stdout = os.Stdout
		cmd.Stderr = os.Stderr
		if err := cmd.Run(); err != nil {
			return fmt.Errorf("build failed in %s: %w", mod, err)
		}
	}
	return nil
}
